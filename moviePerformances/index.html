Movie Performances

  1. oscar nominees budget 

    SELECT title, budget_usd 
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  2. oscar nominees opening week gross 

    SELECT title, opening_week_gross_usd
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  3. oscar nominees latest gross 

    SELECT title, latest_gross_usd
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  4. oscar nominees revenue 

    SELECT title, (latest_gross_usd - budget_usd) AS revenue
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  5. oscar nominees revenue by budget percentage 

    SELECT title, ( (latest_gross_usd - budget_usd) * 100 / budget_usd ) AS percentage 
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  6. oscar nominees budget in global top 100 

    SELECT 
    title, 
    budget_usd, 
    (
        CASE 
	        WHEN id IN (
	            SELECT DISTINCT(movie_id) FROM oscar_movie_award
	        ) 
	        THEN 1 ELSE 0
        END
    ) AS is_oscar
FROM movie
ORDER BY budget_usd DESC
LIMIT 100

  7. oscar nominees revenue in global top 100 

    SELECT 
    title, 
    (latest_gross_usd - budget_usd) AS revenue, 
    (
        CASE 
            WHEN id IN (
                SELECT DISTINCT(movie_id) FROM oscar_movie_award
            ) 
            THEN 1 ELSE 0
        END
    ) AS is_oscar
FROM movie
WHERE latest_gross_usd IS NOT NULL 
ORDER BY revenue DESC
LIMIT 100
