Movie Performances

  1. oscar nominees budget 

    SELECT title, budget_usd 
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  2. oscar nominees opening week gross 

    SELECT title, opening_week_gross_usd
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  3. oscar nominees latest gross 

    SELECT title, latest_gross_usd
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  4. oscar nominees revenue 

    SELECT title, (latest_gross_usd - budget_usd) AS revenue
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)

  5. oscar nominees revenue by budget percentage 

    SELECT title, ( (latest_gross_usd - budget_usd) * 100 / budget_usd ) AS percentage 
FROM movie
WHERE id IN ( 
    SELECT DISTINCT(movie_id) 
    FROM oscar_movie_award
)
